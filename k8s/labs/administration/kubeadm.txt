# Tokens
kubeadm token generate
kubeadm token create --ttl 2h --print-join-command
kubeadm token list
kubeadm token delete <token>

# Join
kubeadm join <api-server>:6443 --token <token> --discovery-token-ca-cert-hash sha256:<hash>

# Certificates
sudo kubeadm certs check-expiration
kubeadm certs certificate-key
sudo kubeadm certs renew all
sudo kubeadm certs renew apiserver

# Configs
kubeadm config print init-defaults > kubeadm-init-defaults.yaml
kubeadm init --config kubeadm-init-config.yaml

kubeadm version
kubeadm config view
